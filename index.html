const SHEET_STUDENTS = 'Students'; const SHEET_BADGES = 'Badges'; const SHEET_EVIDENCE = 'Evidence'; /** * FORCE AUTHORIZATION FUNCTION * * INSTRUCTIONS: * 1. Open this script in your Google Apps Script editor. * 2. In the toolbar at the top, select "triggerAuthorization" from the function dropdown. * 3. Click the "Run" button. * 4. This will force Google to show you the "Authorization Required" popup window! * 5. Grant all permissions (click Advanced -> Go to Untitled Project -> Allow). */ function triggerAuthorization() { Logger.log("Forcing Google Drive & Sheets permissions..."); try { // Call DriveApp to trigger authorization dialog const folders = DriveApp.getFoldersByName("Lab Tracker Evidence"); Logger.log("Drive Access Authorized successfully! Folders found: " + folders.hasNext()); // Call SpreadsheetApp to verify sheets access const ss = SpreadsheetApp.getActiveSpreadsheet(); Logger.log("Spreadsheet Access Authorized successfully! Name: " + ss.getName()); } catch (e) { Logger.log("Authorization trigger run failed: " + e.toString()); } } function doGet(e) { try { const action = e.parameter.action; if (action === 'getStudents') { return createJsonResponse({ status: 'success', data: getStudents() }); } else if (action === 'getBadges') { return createJsonResponse({ status: 'success', data: getBadges() }); } else { return HtmlService.createHtmlOutputFromFile('index') .setTitle('Lab Tracker') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL) .addMetaTag('viewport', 'width=device-width, initial-scale=1'); } } catch (error) { return createJsonResponse({ status: 'error', message: error.toString() }); } } function getScriptUrl() { return ScriptApp.getService().getUrl(); } function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const action = payload.action; if (action === 'postEvidence') { const result = postEvidence( payload.studentId, payload.badgeId, payload.voiceTranscript, payload.photoData // Receive image base64 ); return createJsonResponse({ status: 'success', data: result }); } else { return createJsonResponse({ status: 'error', message: 'Invalid or missing action.' }); } } catch (error) { return createJsonResponse({ status: 'error', message: error.toString() }); } } function getStudents() { return getSheetDataAsObjects(SHEET_STUDENTS); } function getBadges() { return getSheetDataAsObjects(SHEET_BADGES); } /** * Appends data to sheet, saves image data to Google Drive, and links it. */ function postEvidence(studentId, badgeId, voiceTranscript, photoData) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_EVIDENCE); if (!sheet) throw new Error("The tab named '" + SHEET_EVIDENCE + "' was not found."); const timestamp = new Date().toISOString(); const evidenceId = Utilities.getUuid(); let finalPhotoLink = ""; // If a photo was taken, save it to Google Drive and get the link if (photoData && (photoData.indexOf("data:image") !== -1 || photoData.length > 100)) { try { // Find or create folder let folder; const folders = DriveApp.getFoldersByName("Lab Tracker Evidence"); if (folders.hasNext()) { folder = folders.next(); } else { folder = DriveApp.createFolder("Lab Tracker Evidence"); } // Convert the Base64 Data URI into a file let contentType = "image/jpeg"; let base64String = photoData; if (photoData.indexOf(",") !== -1) { const splitData = photoData.split(","); contentType = splitData[0].match(/:(.*?);/)[1]; base64String = splitData[1]; } const bytes = Utilities.base64Decode(base64String); const cleanStudentId = studentId ? studentId.replace(/[^a-zA-Z0-9]/g, "_") : "Unknown"; const cleanBadgeId = badgeId ? badgeId.replace(/[^a-zA-Z0-9]/g, "_") : "Unknown"; const fileName = cleanStudentId + "-" + cleanBadgeId + "-" + Date.now() + ".jpg"; const blob = Utilities.newBlob(bytes, contentType, fileName); // Save file and set it so anyone with the link can view it const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); finalPhotoLink = file.getUrl(); } catch (imageError) { // Return error description in cell so we know exactly why Drive failed finalPhotoLink = "Drive Upload Error: " + imageError.toString(); } } else { finalPhotoLink = photoData ? "Photo Data Malformed or Empty" : "No Photo Attached"; } // Appends: [ID, Timestamp, Student ID, Badge ID, Transcript, Clickable Photo Link] sheet.appendRow([evidenceId, timestamp, studentId, badgeId, voiceTranscript, finalPhotoLink]); return { evidenceId: evidenceId, timestamp: timestamp, photoUrl: finalPhotoLink, message: "Evidence successfully saved!" }; } function getSheetDataAsObjects(sheetName) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) throw new Error("Tab '" + sheetName + "' not found."); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const headers = data[0]; const rows = data.slice(1); return rows.map(row => { let obj = {}; headers.forEach((header, index) => { obj[header] = row[index]; }); return obj; }); } function createJsonResponse(responseObject) { return ContentService.createTextOutput(JSON.stringify(responseObject)).setMimeType(ContentService.MimeType.JSON); }
